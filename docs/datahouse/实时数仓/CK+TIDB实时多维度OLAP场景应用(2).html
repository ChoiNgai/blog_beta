insert into default.sl_dm_store_behavior_session_ps_distributed6
select dt,app_id,row_key,store_id,ua_type,ua_browser,ua_os_version,ua_os,city,province,region,pv,uv,visit_checkout_cnt,completed_checkout_cnt

-------------物化视图-------------
CREATE MATERIALIZED VIEW default.sl_dm_store_behavior_session_local6_summing on cluster cluster_01
to (
    dt                     DateTime,
    app_id                 String,
    store_id               Int64,
    ua_type                String,
    ua_browser             String,
    ua_os_version          String,
    ua_os                  String,
    city                   String,
    province               String,
    region                 String,
    pv                     Int64,
    uv                     Int32,
    visit_checkout_cnt     Int32,
    completed_checkout_cnt Int32
)

ENGINE = ReplicatedSummingMergeTree('/clickhouse/tables/{layer}-{shard}/sl_dm_store_behavior_session_local6_summing', '{replica}')
PARTITION BY (store_id,toYear(dt)) ORDER BY (dt,app_id,ua_type,ua_browser,ua_os_version,ua_os,city,province,region)
POPULATE AS SELECT
  toStartOfHour(dt) as dt,app_id,store_id,
          ifNull(ua_type,'N/A') as ua_type,
          ifNull(ua_browser,'N/A') as ua_browser,
          ifNull(ua_os_version,'N/A') as ua_os_version,
          ifNull(ua_os,'N/A') as ua_os,
          ifNull(city,'N/A') as city,
          ifNull(province,'N/A') as province,
          ifNull(region,'N/A') as region,
  sum(pv) as pv,sum(uv) as uv,sum(visit_checkout_cnt) as visit_checkout_cnt,sum(completed_checkout_cnt) as completed_checkout_cnt
FROM default.sl_dm_store_behavior_session_ps_local6
GROUP BY store_id,dt,app_id,ua_type,ua_browser,ua_os_version,ua_os,city,province,region;


CREATE TABLE default.sl_dm_store_behavior_session_distributed6_summing ON CLUSTER cluster_01 AS default.sl_dm_store_behavior_session_local6_summing
ENGINE = Distributed('cluster_01', 'default', 'sl_dm_store_behavior_session_local6_summing', store_id); 

optimize table default.sl_dm_store_behavior_session_ps_local6;

select dt,
    app_id,
    store_id,
    ua_type,
    ua_browser,
    pv,
    uv,
    visit_checkout_cnt,
    completed_checkout_cnt
from
(
    (SELECT
          dt,
          app_id,
          store_id,
          ua_type,
          ua_browser,
          sum(pv) AS pv,
          sum(uv) AS uv,
          sum(visit_checkout_cnt) AS visit_checkout_cnt,
          sum(completed_checkout_cnt) AS completed_checkout_cnt
      FROM default.sl_dm_store_behavior_session_distributed10
      WHERE (store_id = ${ramdom_ten}) AND (dt < toDateTime('2021-08-01'))
      GROUP BY
        store_id,
        dt,
        app_id,
        ua_type,
        ua_browser
      ORDER BY uv DESC
      LIMIT 1000)
      union all
      (select dt,app_id,store_id,ua_type,ua_browser,
          sum(pv) pv,
          sum(uv) uv,
          sum(visit_checkout_cnt) visit_checkout_cnt,
          sum(completed_checkout_cnt) completed_checkout_cnt
      from
          (select dt,app_id,store_id,ua_type,ua_browser,pv,uv,visit_checkout_cnt,completed_checkout_cnt
          from mysql_db.sl_dm_store_behavior_session_hourly_rt where store_id=${ramdom_ten} and dt>=toDateTime('2021-08-20')
          )
      group by  store_id,dt,app_id,ua_type,ua_browser
      order by uv desc limit 1000
      )
)
ORDER BY uv DESC
LIMIT 1000;

# 结论

(1) 4.1和4.3是利用物化视图后减少了聚合，然后使用不变的uv作为排序健从而提升性能，缺点是如果UV值改变需要重写分区，而且如果使用非UV排序，性能类似3.1. ​

(2)建立了最细粒度的物化视图后，再做聚合，如果聚合后的总数还是很多（类似4.4），那么性能还是会很低，这时候建议再建一个物化视图。但是如果再建物化视图又衍生一个问题，上层的应用需要人为指定读哪个视图。 ​

(3)4.2读取大量的tidb数据时，会导致tidb读过热，tidb+tikv CPU同时出现过热的情况，查询时即使group sum limit等方式也不会下推给tidb执行。 tidb cpu过热请看下图 jmeter通过调用clickhouse聚合tidb数据 tidb慢日志
