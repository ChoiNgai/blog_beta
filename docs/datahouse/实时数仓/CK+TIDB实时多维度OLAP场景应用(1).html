# 1.哪些店铺需要同步到CK

统计规则（ck统计，并排除已经在同步表的店铺）： （1）当天数据量>50w 或 （2）最近一个月数据量>100w

# 2.如何将tidb数据同步到CK

使用ck原生SQL写入到ck分布式表 举例：

insert into default.sl_dm_store_behavior_session_distributed
select dt,app_id,row_key,store_id,ua_type,ua_browser,ua_os_version,
ua_os,city,province,region,pv,uv,visit_checkout_cnt,completed_checkout_cnt 
from mysql_db.sl_dm_store_behavior_session_hourly_rt
where dt>=toDateTime('2021-08-01') and dt<oDateTime('2021-08-02);

CK分布式全量表（随机分片+dt分区）

create table default.sl_dm_store_behavior_session_local on cluster cluster_01
(
    dt                     DateTime,
    app_id                 String,
    row_key                String,
    store_id               Int64,
    ua_type                Nullable(String),
    ua_browser             Nullable(String),
    ua_os_version          Nullable(String),
    ua_os                  Nullable(String),
    city                   Nullable(String),
    province               Nullable(String),
    region                 Nullable(String),
    pv                     Nullable(Int64),
    uv                     Nullable(Int32),
    visit_checkout_cnt     Nullable(Int32),
    completed_checkout_cnt Nullable(Int32)
)
    ENGINE = ReplicatedReplacingMergeTree('/clickhouse/tables/{layer}-{shard}/sl_dm_store_behavior_session_local', '{replica}')
    PARTITION BY toYYYYMM(dt)
    ORDER BY (store_id,dt,row_key)
    primary key (store_id,dt,row_key);    

CREATE TABLE default.sl_dm_store_behavior_session_distributed ON CLUSTER cluster_01 AS default.sl_dm_store_behavior_session_local
ENGINE = Distributed(cluster_01, default, sl_dm_store_behavior_session_local, rand());

CK店铺表（store_id分片+store_id+月分区）

create table default.sl_dm_store_behavior_session_local_store_id on cluster cluster_01
(
    dt                     DateTime,
    app_id                 String,
    store_id               Int64,
    ua_type                String,
    ua_browser             String,
    ua_os_version          String,
    ua_os                  String,
    city                   String,
    province               String,
    region                 String,
    pv                     Int64,
    uv                     Int32,
    visit_checkout_cnt     Int32,
    completed_checkout_cnt Int32,
    INDEX tag_idx_store_id_dt(store_id,dt) TYPE minmax GRANULARITY 1
)
ENGINE = ReplicatedReplacingMergeTree('/clickhouse/tables/{layer}-{shard}/sl_dm_store_behavior_session_local_store_id', '{replica}')
PARTITION BY (store_id,toYYYYMM(dt))
ORDER BY (uv,store_id,dt,app_id,ua_type,ua_browser,ua_os_version,ua_os,city,province,region);

CREATE TABLE default.sl_dm_store_behavior_session_distributed_store_id ON CLUSTER cluster_01 AS default.sl_dm_store_behavior_session_local_store_id
ENGINE = Distributed('cluster_01', 'default', 'sl_dm_store_behavior_session_local_store_id', store_id);

"CK分布式全量表"写入到"CK店铺表"SQL

insert into default.sl_dm_store_behavior_session_distributed10
select toStartOfHour(dt) as dt,app_id,store_id,
       ifNull(ua_type,'N/A') as ua_type,
       ifNull(ua_browser,'N/A') as ua_browser,
       ifNull(ua_os_version,'N/A') as ua_os_version,
       ifNull(ua_os,'N/A') as ua_os,
       ifNull(city,'N/A') as city,
       ifNull(province,'N/A') as province,
       ifNull(region,'N/A') as region,
  sum(pv) as pv,sum(uv) as uv,sum(visit_checkout_cnt) as visit_checkout_cnt,sum(completed_checkout_cnt) as completed_checkout_cnt
from default.sl_dm_store_behavior_session_local l join 
tidb店铺同步表
where store_id<=10 and dt=当天时间范围
GROUP BY store_id,dt,app_id,ua_type,ua_browser,ua_os_version,ua_os,city,province,region;
