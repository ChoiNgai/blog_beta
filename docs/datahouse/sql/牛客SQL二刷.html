# 21

SELECT emp_no,(EndSalary- StartSalary) AS growth FROM (
    SELECT t1.emp_no,t2.salary AS EndSalary,t3.salary AS StartSalary
    FROM salaries t1
    LEFT JOIN (
        SELECT emp_no,salary FROM salaries
        WHERE to_date='9999-01-01'
    ) t2 ON t1.emp_no = t2.emp_no
    LEFT JOIN (
        SELECT salaries.emp_no,salary FROM salaries
        LEFT JOIN employees ON salaries.emp_no = salaries.emp_no
        WHERE from_date = hire_date
        GROUP BY salaries.emp_no
    ) t3 ON t2.emp_no = t3.emp_no
    WHERE t1.emp_no IN (
        -- 查询在职员工编号
        SELECT emp_no FROM salaries
        WHERE to_date='9999-01-01'
    )
    GROUP BY t1.emp_no
) t
ORDER BY (EndSalary- StartSalary)

错误写法（这种没考虑到降薪的情况）：

SELECT emp_no,(MaxSalary- MinSalary) AS growth FROM (
    SELECT emp_no,MAX(salary) AS MaxSalary,MIN(salary)  AS MinSalary
    FROM salaries
    WHERE emp_no IN (
        -- 查询在职员工编号
        SELECT emp_no FROM salaries
        WHERE to_date='9999-01-01'
    )
    GROUP BY emp_no
) t
ORDER BY (MaxSalary- MinSalary)
